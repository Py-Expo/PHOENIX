var questions = [];

$.ajax({
    url: 'https://opentdb.com/api.php?amount=10&category=18&type=multiple',
    method: 'GET',
    success: function(response) {
        var apiQuestions = response.results;

        apiQuestions.forEach(function(apiQuestion) {
            var question = {
                question: apiQuestion.question,
                options: apiQuestion.incorrect_answers.concat(apiQuestion.correct_answer), // Combine incorrect and correct answers
                answer: apiQuestion.incorrect_answers.length // Index of correct answer in the options array
            };
            questions.push(question);
        });

        if (questions.length === 0) {
            console.error('No questions retrieved from the API.');
            return;
        }
    },
    error: function(xhr, status, error) {
        console.error('Error fetching questions:', error);
        alert("Failed to fetch questions. Please try again later.");
    }
});

$(document).ready(function() {
    var email = "";
    $("#quiz-container").hide();
    $("#result").hide();

     // Login handling
  $("#login-form").submit(function(event) {
    event.preventDefault();
    const username = $("#username").val();
    const password = $("#password").val();

    $.ajax({
        url: '/getuser/',
        success: function (data) {
          var userData;
          debugger;

        // Assuming 'username' and 'password' are already defined somewhere in your code
        var username = $("#username").val().toLowerCase();
        var password = $("#password").val();
        var validUserDetail = null;
        // Loop through each user data object
        var validUser = data.some(function(user) {
          if (user.firstname.toLowerCase() === username && user.password === password) {
            validUserDetail = user;
            return true; // Stop the iteration once a valid user is found
          }
        });

        if (validUserDetail !== null && validUser !== false) {
            email = validUserDetail.email;
            $("#username").val("");
            $("#password").val("");
            $("#login-container").hide();
            $("#quiz-container").show();
            displayQuestion();
        } else {
            alert("Invalid username or password!");
        }
      },
      error: function (xhr, status, error) {
          console.error('Request failed. Status: ' + status);
      }
    });
  });
  
    let currentQuestion = 0;
    let score = 0;
  
    function displayQuestion() {
      if (currentQuestion >= questions.length) {
        showResult();
      }
  
      const question = questions[currentQuestion];
      $("#question").text(question.question);
      $("#options").empty();
      $("#feedback").hide();
  
      question.options.forEach((option, index) => {
        const optionElement = $('<li><input type="radio" name="answer" value="' + index + '">' + option + '</li>');
        $("#options").append(optionElement);
      });
    }
  
    function checkAnswer() {
      const selectedOption = $("input[name='answer']:checked").val();
      if (selectedOption !== undefined) {
        const correctAnswer = questions[currentQuestion].answer;
        if (selectedOption == correctAnswer) {
          score++;
        } else {
          score -= 0.25;
        }
        currentQuestion++;
        displayQuestion();
      }else{
        alert("Choose any answer");
      }
    }
  
    function showResult() {
      $("#question").text("Quiz Completed!");
      $("#options").empty();
      $("#next-question").hide();
      $("#result").show().text("Your score: " + score + " out of " + questions.length);
      debugger;
      $.ajax({
        url: '/send-email/',
        data: {
            'score': score, // Example score, replace with actual data
            'email': email // Example email, replace with actual data
      },
      success: function(response) {
        debugger;
          console.log(response);
          // Handle success response
      },
      error: function(xhr, status, error) {
        debugger;
          console.error('Request failed. Status: ' + status);
      }
});

    }
    if(questions.length != 0){
      displayQuestion();
    }

    $("#next-question").click(checkAnswer);
});
